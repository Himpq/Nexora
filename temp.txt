#!/usr/bin/env python3
"""
Quick StepFun Search test script.

Usage:
  1) Set env var:
     - PowerShell: $env:STEPFUN_API_KEY = "your_key"
  2) Run:
     python temp.txt
     python temp.txt "Beijing weather"
"""

import json
import os
import sys
import urllib.error
import urllib.request


def main() -> int:
    api_key = os.getenv("STEPFUN_API_KEY", "").strip()
    if not api_key:
        print("ERROR: STEPFUN_API_KEY is empty.")
        print('PowerShell: $env:STEPFUN_API_KEY = "your_key"')
        return 1

    query = "北京 天气"
    if len(sys.argv) > 1 and sys.argv[1].strip():
        query = sys.argv[1].strip()

    payload = {
        "query": query,
        "n": 1,
        "category": "research",
    }
    body = json.dumps(payload, ensure_ascii=False).encode("utf-8")

    req = urllib.request.Request(
        url="https://api.stepfun.com/v1/search",
        data=body,
        method="POST",
        headers={
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json; charset=utf-8",
        },
    )

    try:
        with urllib.request.urlopen(req, timeout=30) as resp:
            raw = resp.read().decode("utf-8", errors="replace")
            try:
                obj = json.loads(raw)
                print(json.dumps(obj, ensure_ascii=False, indent=2))
            except Exception:
                print(raw)
        return 0
    except urllib.error.HTTPError as e:
        err = e.read().decode("utf-8", errors="replace")
        print(f"HTTP {e.code}: {err}")
        return 2
    except Exception as e:
        print(f"Request failed: {e}")
        return 3


if __name__ == "__main__":
    raise SystemExit(main())
